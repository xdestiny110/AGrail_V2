function readPB()
    pb = require("luapb")
    print("load lua-protobuf")
    local p =""
    if(CS.Framework.Framework.AssetBundle.AssetBundleManager.SimulationMode)
    then
        p = CS.Framework.Tool.UnityPathToSystemPath("Assets/ProtoFile/base.pb")
    else
        p = CS.Framework.PlatformPath.PersistFileSysPath("base.pb")
    end
    assert(pb.loadfile(p))
    print(string.format("load pb from %s", p))

    local loginReq = {
        asGuest = false,
        user_id = "Xdestiny",
        user_password = "123456"
    }
    local en = assert(pb.encode("AGrailPB.LoginRequest", loginReq))
    local de = assert(pb.decode("AGrailPB.LoginRequest", en))
    print(de.user_id)
end
